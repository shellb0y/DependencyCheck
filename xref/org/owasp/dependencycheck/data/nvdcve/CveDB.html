214     /**
215      * Tries to determine the product name of the database.
216      *
217      * @param conn the database connection
218      * @return the product name of the database if successful, {@code null} else
219      */
220     private String determineDatabaseProductName(Connection conn) {
221         try {
222             final String databaseProductName = conn.getMetaData().getDatabaseProductName().toLowerCase();
223             LOGGER.debug("Database product: {}", databaseProductName);
224             return databaseProductName;
225         } catch (SQLException se) {
226             LOGGER.warn("Problem determining database product!", se);
227             return null;

231     /**
232      * Opens the database connection. If the database does not exist, it will
233      * create a new one.
234      *
235      * @throws DatabaseException thrown if there is an error opening the
236      * database connection
237      */
238     private synchronized void open() throws DatabaseException {
239         try {
240             if (!isOpen()) {
241                 connection = connectionFactory.getConnection();
242                 final String databaseProductName = determineDatabaseProductName(this.connection);
243                 statementBundle = databaseProductName != null
244                         ? ResourceBundle.getBundle("data/dbStatements", new Locale(databaseProductName))
245                         : ResourceBundle.getBundle("data/dbStatements");
246                 prepareStatements();
247                 databaseProperties = new DatabaseProperties(this);
248             }
249         } catch (DatabaseException e) {
250             releaseResources();
251             throw e;
252         }
253     }
254 
255     /**
256      * Closes the database connection. Close should be called on this object
257      * when it is done being used.
258      */
259     @Override
260     public synchronized void close() {
261         if (isOpen()) {
262             clearCache();
263             closeStatements();
264             try {
265                 connection.close();
266             } catch (SQLException ex) {
267                 LOGGER.error("There was an error attempting to close the CveDB, see the log for more details.");
268                 LOGGER.debug("", ex);
269             } catch (Throwable ex) {
270                 LOGGER.error("There was an exception attempting to close the CveDB, see the log for more details.");
271                 LOGGER.debug("", ex);
272             }
273             releaseResources();
274             connectionFactory.cleanup();
275         }
276     }

297     /**
298      * Prepares all statements to be used.
299      *
300      * @throws DatabaseException thrown if there is an error preparing the
301      * statements
302      */
303     private void prepareStatements() throws DatabaseException {
304         for (PreparedStatementCveDb key : values()) {
305             PreparedStatement preparedStatement = null;
306             try {
307                 final String statementString = statementBundle.getString(key.name());
308                 if (key == INSERT_VULNERABILITY || key == INSERT_CPE) {
309                     preparedStatement = connection.prepareStatement(statementString, new String[]{"id"});
310                 } else {
311                     preparedStatement = connection.prepareStatement(statementString);
312                 }
313             } catch (SQLException ex) {
314                 throw new DatabaseException(ex);
315             } catch (MissingResourceException ex) {
316                 if (!ex.getMessage().contains("key MERGE_PROPERTY")) {
317                     throw new DatabaseException(ex);
318                 }
319             }
320             if (preparedStatement != null) {
321                 preparedStatements.put(key, preparedStatement);
322             }
323         }
324     }
325 
326     /**
327      * Closes all prepared statements.
328      */
329     private synchronized void closeStatements() {
330         for (PreparedStatement preparedStatement : preparedStatements.values()) {
331             DBUtils.closeStatement(preparedStatement);
332         }
333     }

335     /**
336      * Returns the specified prepared statement.
337      *
338      * @param key the prepared statement from {@link PreparedStatementCveDb} to
339      * return
340      * @return the prepared statement
341      * @throws SQLException thrown if a SQL Exception occurs
342      */
343     private synchronized PreparedStatement getPreparedStatement(PreparedStatementCveDb key) throws SQLException {
344         if (!preparedStatements.containsKey(key)) {
345             return null;
346         }
347         final PreparedStatement preparedStatement = preparedStatements.get(key);
348         preparedStatement.clearParameters();
349         return preparedStatement;
350     }

397     /**
398      * Searches the CPE entries in the database and retrieves all entries for a
399      * given vendor and product combination. The returned list will include all
400      * versions of the product that are registered in the NVD CVE data.
401      *
402      * @param vendor the identified vendor name of the dependency being analyzed
403      * @param product the identified name of the product of the dependency being
404      * analyzed
405      * @return a set of vulnerable software
406      */
407     public synchronized Set<VulnerableSoftware> getCPEs(String vendor, String product) {
408         final Set<VulnerableSoftware> cpe = new HashSet<>();
409         ResultSet rs = null;
410         try {
411             final PreparedStatement ps = getPreparedStatement(SELECT_CPE_ENTRIES);
412             ps.setString(1, vendor);
413             ps.setString(2, product);
414             rs = ps.executeQuery();
415 
416             while (rs.next()) {
417                 final VulnerableSoftware vs = new VulnerableSoftware();
418                 vs.setCpe(rs.getString(1));
419                 cpe.add(vs);
420             }
421         } catch (SQLException ex) {
422             LOGGER.error("An unexpected SQL Exception occurred; please see the verbose log for more details.");
423             LOGGER.debug("", ex);
424         } finally {
425             DBUtils.closeResultSet(rs);
426         }
427         return cpe;
428     }

478     /**
479      * Saves a property to the database.
480      *
481      * @param key the property key
482      * @param value the property value
483      */
484     public synchronized void saveProperty(String key, String value) {
485         clearCache();
486         try {
487             final PreparedStatement mergeProperty = getPreparedStatement(MERGE_PROPERTY);
488             if (mergeProperty != null) {
489                 mergeProperty.setString(1, key);
490                 mergeProperty.setString(2, value);
491                 mergeProperty.execute();
492             } else {
493                 // No Merge statement, so doing an Update/Insert...
494                 final PreparedStatement updateProperty = getPreparedStatement(UPDATE_PROPERTY);
495                 updateProperty.setString(1, value);
496                 updateProperty.setString(2, key);
497                 if (updateProperty.executeUpdate() == 0) {
498                     final PreparedStatement insertProperty = getPreparedStatement(INSERT_PROPERTY);
499                     insertProperty.setString(1, key);
500                     insertProperty.setString(2, value);
501                     insertProperty.executeUpdate();
502                 }
503             }
504         } catch (SQLException ex) {
505             LOGGER.warn("Unable to save property '{}' with a value of '{}' to the database", key, value);
506             LOGGER.debug("", ex);
507         }
508     }

522     /**
523      * Retrieves the vulnerabilities associated with the specified CPE.
524      *
525      * @param cpeStr the CPE name
526      * @return a list of Vulnerabilities
527      * @throws DatabaseException thrown if there is an exception retrieving data
528      */
529     public synchronized List<Vulnerability> getVulnerabilities(String cpeStr) throws DatabaseException {
530         final List<Vulnerability> cachedVulnerabilities = vulnerabilitiesForCpeCache.get(cpeStr);
531         if (cachedVulnerabilities != null) {
532             LOGGER.debug("Cache hit for {}", cpeStr);
533             return cachedVulnerabilities;
534         } else {
535             LOGGER.debug("Cache miss for {}", cpeStr);
536         }
537         final VulnerableSoftware cpe = new VulnerableSoftware();
538         try {
539             cpe.parseName(cpeStr);
540         } catch (UnsupportedEncodingException ex) {
541             LOGGER.trace("", ex);
542         }
543         final DependencyVersion detectedVersion = parseDependencyVersion(cpe);
544         final List<Vulnerability> vulnerabilities = new ArrayList<>();
545 
546         ResultSet rs = null;
547         try {
548             final PreparedStatement ps = getPreparedStatement(SELECT_CVE_FROM_SOFTWARE);
549             ps.setString(1, cpe.getVendor());
550             ps.setString(2, cpe.getProduct());
551             rs = ps.executeQuery();
552             String currentCVE = "";
553 
554             final Map<String, Boolean> vulnSoftware = new HashMap<>();
555             while (rs.next()) {
556                 final String cveId = rs.getString(1);
557                 if (!currentCVE.equals(cveId)) { //check for match and add
558                     final Entry<String, Boolean> matchedCPE = getMatchingSoftware(vulnSoftware, cpe.getVendor(), cpe.getProduct(), detectedVersion);
559                     if (matchedCPE != null) {
560                         final Vulnerability v = getVulnerability(currentCVE);
561                         if (v != null) {
562                             v.setMatchedCPE(matchedCPE.getKey(), matchedCPE.getValue() ? "Y" : null);
563                             vulnerabilities.add(v);
564                         }
565                     }
566                     vulnSoftware.clear();
567                     currentCVE = cveId;
568                 }
569 
570                 final String cpeId = rs.getString(2);
571                 final String previous = rs.getString(3);
572                 final Boolean p = previous != null && !previous.isEmpty();
573                 vulnSoftware.put(cpeId, p);
574             }
575             //remember to process the last set of CVE/CPE entries
576             final Entry<String, Boolean> matchedCPE = getMatchingSoftware(vulnSoftware, cpe.getVendor(), cpe.getProduct(), detectedVersion);
577             if (matchedCPE != null) {
578                 final Vulnerability v = getVulnerability(currentCVE);
579                 if (v != null) {
580                     v.setMatchedCPE(matchedCPE.getKey(), matchedCPE.getValue() ? "Y" : null);
581                     vulnerabilities.add(v);
582                 }
583             }
584         } catch (SQLException ex) {
585             throw new DatabaseException("Exception retrieving vulnerability for " + cpeStr, ex);
586         } finally {
587             DBUtils.closeResultSet(rs);
588         }
589         vulnerabilitiesForCpeCache.put(cpeStr, vulnerabilities);
590         return vulnerabilities;
591     }

593     /**
594      * Gets a vulnerability for the provided CVE.
595      *
596      * @param cve the CVE to lookup
597      * @return a vulnerability object
598      * @throws DatabaseException if an exception occurs
599      */
600     public synchronized Vulnerability getVulnerability(String cve) throws DatabaseException {
601         ResultSet rsV = null;
602         ResultSet rsR = null;
603         ResultSet rsS = null;
604         Vulnerability vuln = null;
605 
606         try {
607             final PreparedStatement psV = getPreparedStatement(SELECT_VULNERABILITY);
608             psV.setString(1, cve);
609             rsV = psV.executeQuery();
610             if (rsV.next()) {
611                 vuln = new Vulnerability();
612                 vuln.setName(cve);
613                 vuln.setDescription(rsV.getString(2));
614                 String cwe = rsV.getString(3);
615                 if (cwe != null) {
616                     final String name = CweDB.getCweName(cwe);
617                     if (name != null) {
618                         cwe += ' ' + name;
619                     }
620                 }
621                 final int cveId = rsV.getInt(1);
622                 vuln.setCwe(cwe);
623                 vuln.setCvssScore(rsV.getFloat(4));
624                 vuln.setCvssAccessVector(rsV.getString(5));
625                 vuln.setCvssAccessComplexity(rsV.getString(6));
626                 vuln.setCvssAuthentication(rsV.getString(7));
627                 vuln.setCvssConfidentialityImpact(rsV.getString(8));
628                 vuln.setCvssIntegrityImpact(rsV.getString(9));
629                 vuln.setCvssAvailabilityImpact(rsV.getString(10));
630 
631                 final PreparedStatement psR = getPreparedStatement(SELECT_REFERENCES);
632                 psR.setInt(1, cveId);
633                 rsR = psR.executeQuery();
634                 while (rsR.next()) {
635                     vuln.addReference(rsR.getString(1), rsR.getString(2), rsR.getString(3));
636                 }
637 
638                 final PreparedStatement psS = getPreparedStatement(SELECT_SOFTWARE);
639                 psS.setInt(1, cveId);
640                 rsS = psS.executeQuery();
641                 while (rsS.next()) {
642                     final String cpe = rsS.getString(1);
643                     final String prevVersion = rsS.getString(2);
644                     if (prevVersion == null) {
645                         vuln.addVulnerableSoftware(cpe);
646                     } else {
647                         vuln.addVulnerableSoftware(cpe, prevVersion);
648                     }
649                 }
650             }
651         } catch (SQLException ex) {
652             throw new DatabaseException("Error retrieving " + cve, ex);
653         } finally {
654             DBUtils.closeResultSet(rsV);
655             DBUtils.closeResultSet(rsR);
656             DBUtils.closeResultSet(rsS);
657         }
658         return vuln;
659     }

661     /**
662      * Updates the vulnerability within the database. If the vulnerability does
663      * not exist it will be added.
664      *
665      * @param vuln the vulnerability to add to the database
666      * @throws DatabaseException is thrown if the database
667      */
668     public synchronized void updateVulnerability(Vulnerability vuln) throws DatabaseException {
669         clearCache();
670         ResultSet rs = null;
671         try {
672             int vulnerabilityId = 0;
673             final PreparedStatement selectVulnerabilityId = getPreparedStatement(SELECT_VULNERABILITY_ID);
674             selectVulnerabilityId.setString(1, vuln.getName());
675             rs = selectVulnerabilityId.executeQuery();
676             if (rs.next()) {
677                 vulnerabilityId = rs.getInt(1);
678                 // first delete any existing vulnerability info. We don't know what was updated. yes, slower but atm easier.
679                 final PreparedStatement deleteReference = getPreparedStatement(DELETE_REFERENCE);
680                 deleteReference.setInt(1, vulnerabilityId);
681                 deleteReference.execute();
682 
683                 final PreparedStatement deleteSoftware = getPreparedStatement(DELETE_SOFTWARE);
684                 deleteSoftware.setInt(1, vulnerabilityId);
685                 deleteSoftware.execute();
686             }
687             DBUtils.closeResultSet(rs);
688 
689             if (vulnerabilityId != 0) {
690                 if (vuln.getDescription().contains("** REJECT **")) {
691                     final PreparedStatement deleteVulnerability = getPreparedStatement(DELETE_VULNERABILITY);
692                     deleteVulnerability.setInt(1, vulnerabilityId);
693                     deleteVulnerability.executeUpdate();
694                 } else {
695                     final PreparedStatement updateVulnerability = getPreparedStatement(UPDATE_VULNERABILITY);
696                     updateVulnerability.setString(1, vuln.getDescription());
697                     updateVulnerability.setString(2, vuln.getCwe());
698                     updateVulnerability.setFloat(3, vuln.getCvssScore());
699                     updateVulnerability.setString(4, vuln.getCvssAccessVector());
700                     updateVulnerability.setString(5, vuln.getCvssAccessComplexity());
701                     updateVulnerability.setString(6, vuln.getCvssAuthentication());
702                     updateVulnerability.setString(7, vuln.getCvssConfidentialityImpact());
703                     updateVulnerability.setString(8, vuln.getCvssIntegrityImpact());
704                     updateVulnerability.setString(9, vuln.getCvssAvailabilityImpact());
705                     updateVulnerability.setInt(10, vulnerabilityId);
706                     updateVulnerability.executeUpdate();
707                 }
708             } else {
709                 final PreparedStatement insertVulnerability = getPreparedStatement(INSERT_VULNERABILITY);
710                 insertVulnerability.setString(1, vuln.getName());
711                 insertVulnerability.setString(2, vuln.getDescription());
712                 insertVulnerability.setString(3, vuln.getCwe());
713                 insertVulnerability.setFloat(4, vuln.getCvssScore());
714                 insertVulnerability.setString(5, vuln.getCvssAccessVector());
715                 insertVulnerability.setString(6, vuln.getCvssAccessComplexity());
716                 insertVulnerability.setString(7, vuln.getCvssAuthentication());
717                 insertVulnerability.setString(8, vuln.getCvssConfidentialityImpact());
718                 insertVulnerability.setString(9, vuln.getCvssIntegrityImpact());
719                 insertVulnerability.setString(10, vuln.getCvssAvailabilityImpact());
720                 insertVulnerability.execute();
721                 try {
722                     rs = insertVulnerability.getGeneratedKeys();
723                     rs.next();
724                     vulnerabilityId = rs.getInt(1);
725                 } catch (SQLException ex) {
726                     final String msg = String.format("Unable to retrieve id for new vulnerability for '%s'", vuln.getName());
727                     throw new DatabaseException(msg, ex);
728                 } finally {
729                     DBUtils.closeResultSet(rs);
730                 }
731             }
732 
733             final PreparedStatement insertReference = getPreparedStatement(INSERT_REFERENCE);
734             for (Reference r : vuln.getReferences()) {
735                 insertReference.setInt(1, vulnerabilityId);
736                 insertReference.setString(2, r.getName());
737                 insertReference.setString(3, r.getUrl());
738                 insertReference.setString(4, r.getSource());
739                 insertReference.execute();
740             }
741 
742             final PreparedStatement insertSoftware = getPreparedStatement(INSERT_SOFTWARE);
743             for (VulnerableSoftware s : vuln.getVulnerableSoftware()) {
744                 int cpeProductId = 0;
745                 final PreparedStatement selectCpeId = getPreparedStatement(SELECT_CPE_ID);
746                 selectCpeId.setString(1, s.getName());
747                 try {
748                     rs = selectCpeId.executeQuery();
749                     if (rs.next()) {
750                         cpeProductId = rs.getInt(1);
751                     }
752                 } catch (SQLException ex) {
753                     throw new DatabaseException("Unable to get primary key for new cpe: " + s.getName(), ex);
754                 } finally {
755                     DBUtils.closeResultSet(rs);
756                 }
757 
758                 if (cpeProductId == 0) {
759                     final PreparedStatement insertCpe = getPreparedStatement(INSERT_CPE);
760                     insertCpe.setString(1, s.getName());
761                     insertCpe.setString(2, s.getVendor());
762                     insertCpe.setString(3, s.getProduct());
763                     insertCpe.executeUpdate();
764                     cpeProductId = DBUtils.getGeneratedKey(insertCpe);
765                 }
766                 if (cpeProductId == 0) {
767                     throw new DatabaseException("Unable to retrieve cpeProductId - no data returned");
768                 }
769 
770                 insertSoftware.setInt(1, vulnerabilityId);
771                 insertSoftware.setInt(2, cpeProductId);
772 
773                 if (s.getPreviousVersion() == null) {
774                     insertSoftware.setNull(3, java.sql.Types.VARCHAR);
775                 } else {
776                     insertSoftware.setString(3, s.getPreviousVersion());
777                 }
778                 try {
779                     insertSoftware.execute();
780                 } catch (SQLException ex) {
781                     if (ex.getMessage().contains("Duplicate entry")) {
782                         final String msg = String.format("Duplicate software key identified in '%s:%s'", vuln.getName(), s.getName());
783                         LOGGER.info(msg, ex);
784                     } else {
785                         throw ex;
786                     }
787                 }
788 
789             }
790         } catch (SQLException ex) {
791             final String msg = String.format("Error updating '%s'", vuln.getName());
792             LOGGER.debug(msg, ex);
793             throw new DatabaseException(msg, ex);
794         } finally {
795             DBUtils.closeResultSet(rs);
796         }
797     }

849     /**
850      * Determines if the given identifiedVersion is affected by the given cpeId
851      * and previous version flag. A non-null, non-empty string passed to the
852      * previous version argument indicates that all previous versions are
853      * affected.
854      *
855      * @param vendor the vendor of the dependency being analyzed
856      * @param product the product name of the dependency being analyzed
857      * @param vulnerableSoftware a map of the vulnerable software with a boolean
858      * indicating if all previous versions are affected
859      * @param identifiedVersion the identified version of the dependency being
860      * analyzed
861      * @return true if the identified version is affected, otherwise false
862      */
863     protected Entry<String, Boolean> getMatchingSoftware(Map<String, Boolean> vulnerableSoftware, String vendor, String product,
864             DependencyVersion identifiedVersion) {
865 
866         final boolean isVersionTwoADifferentProduct = "apache".equals(vendor) && "struts".equals(product);
867 
868         final Set<String> majorVersionsAffectingAllPrevious = new HashSet<>();
869         final boolean matchesAnyPrevious = identifiedVersion == null || "-".equals(identifiedVersion.toString());
870         String majorVersionMatch = null;
871         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
872             final DependencyVersion v = parseDependencyVersion(entry.getKey());
873             if (v == null || "-".equals(v.toString())) { //all versions
874                 return entry;
875             }
876             if (entry.getValue()) {
877                 if (matchesAnyPrevious) {
878                     return entry;
879                 }
880                 if (identifiedVersion != null && identifiedVersion.getVersionParts().get(0).equals(v.getVersionParts().get(0))) {
881                     majorVersionMatch = v.getVersionParts().get(0);
882                 }
883                 majorVersionsAffectingAllPrevious.add(v.getVersionParts().get(0));
884             }
885         }
886         if (matchesAnyPrevious) {
887             return null;
888         }
889 
890         final boolean canSkipVersions = majorVersionMatch != null && majorVersionsAffectingAllPrevious.size() > 1;
891         //yes, we are iterating over this twice. The first time we are skipping versions those that affect all versions
892         //then later we process those that affect all versions. This could be done with sorting...
893         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
894             if (!entry.getValue()) {
895                 final DependencyVersion v = parseDependencyVersion(entry.getKey());
896                 //this can't dereference a null 'majorVersionMatch' as canSkipVersions accounts for this.
897                 if (canSkipVersions && majorVersionMatch != null && !majorVersionMatch.equals(v.getVersionParts().get(0))) {
898                     continue;
899                 }
900                 //this can't dereference a null 'identifiedVersion' because if it was null we would have exited
901                 //in the above loop or just after loop (if matchesAnyPrevious return null).
902                 if (identifiedVersion != null && identifiedVersion.equals(v)) {
903                     return entry;
904                 }
905             }
906         }
907         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
908             if (entry.getValue()) {
909                 final DependencyVersion v = parseDependencyVersion(entry.getKey());
910                 //this can't dereference a null 'majorVersionMatch' as canSkipVersions accounts for this.
911                 if (canSkipVersions && majorVersionMatch != null && !majorVersionMatch.equals(v.getVersionParts().get(0))) {
912                     continue;
913                 }
914                 //this can't dereference a null 'identifiedVersion' because if it was null we would have exited
915                 //in the above loop or just after loop (if matchesAnyPrevious return null).
916                 if (entry.getValue() && identifiedVersion != null && identifiedVersion.compareTo(v) <= 0
917                         && !(isVersionTwoADifferentProduct && !identifiedVersion.getVersionParts().get(0).equals(v.getVersionParts().get(0)))) {
918                     return entry;
919                 }
920             }
921         }
922         return null;
923     }

925     /**
926      * Parses the version (including revision) from a CPE identifier. If no
927      * version is identified then a '-' is returned.
928      *
929      * @param cpeStr a cpe identifier
930      * @return a dependency version
931      */
932     private DependencyVersion parseDependencyVersion(String cpeStr) {
933         final VulnerableSoftware cpe = new VulnerableSoftware();
934         try {
935             cpe.parseName(cpeStr);
936         } catch (UnsupportedEncodingException ex) {
937             //never going to happen.
938             LOGGER.trace("", ex);
939         }
940         return parseDependencyVersion(cpe);
941     }

943     /**
944      * Takes a CPE and parses out the version number. If no version is
945      * identified then a '-' is returned.
946      *
947      * @param cpe a cpe object
948      * @return a dependency version
949      */
950     private DependencyVersion parseDependencyVersion(VulnerableSoftware cpe) {
951         final DependencyVersion cpeVersion;
952         if (cpe.getVersion() != null && !cpe.getVersion().isEmpty()) {
953             final String versionText;
954             if (cpe.getUpdate() != null && !cpe.getUpdate().isEmpty()) {
955                 versionText = String.format("%s.%s", cpe.getVersion(), cpe.getUpdate());
956             } else {
957                 versionText = cpe.getVersion();
958             }
959             cpeVersion = DependencyVersionUtil.parseVersion(versionText);
960         } else {
961             cpeVersion = new DependencyVersion("-");
962         }
963         return cpeVersion;
964     }
