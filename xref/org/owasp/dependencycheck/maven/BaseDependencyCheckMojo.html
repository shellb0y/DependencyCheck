562     /**
563      * Executes dependency-check.
564      *
565      * @throws MojoExecutionException thrown if there is an exception executing
566      * the mojo
567      * @throws MojoFailureException thrown if dependency-check failed the build
568      */
569     @Override
570     public void execute() throws MojoExecutionException, MojoFailureException {
571         generatingSite = false;
572         if (skip) {
573             getLog().info("Skipping " + getName(Locale.US));
574         } else {
575             validateAggregate();
576             project.setContextValue(getOutputDirectoryContextKey(), this.outputDirectory);
577             runCheck();
578         }
579     }
580 
581     /**
582      * Checks if the aggregate configuration parameter has been set to true. If
583      * it has a MojoExecutionException is thrown because the aggregate
584      * configuration parameter is no longer supported.
585      *
586      * @throws MojoExecutionException thrown if aggregate is set to true
587      */
588     private void validateAggregate() throws MojoExecutionException {
589         if (aggregate != null && aggregate) {
590             final String msg = "Aggregate configuration detected - as of dependency-check 1.2.8 this no longer supported. "
591                     + "Please use the aggregate goal instead.";
592             throw new MojoExecutionException(msg);
593         }
594     }

639     /**
640      * Generates the Dependency-Check Site Report.
641      *
642      * @param sink the sink to write the report to
643      * @param locale the locale to use when generating the report
644      * @throws MavenReportException if a maven report exception occurs
645      */
646     public void generate(Sink sink, Locale locale) throws MavenReportException {
647         if (skip) {
648             getLog().info("Skipping report generation " + getName(Locale.US));
649             return;
650         }
651 
652         generatingSite = true;
653         try {
654             validateAggregate();
655         } catch (MojoExecutionException ex) {
656             throw new MavenReportException(ex.getMessage());
657         }
658         project.setContextValue(getOutputDirectoryContextKey(), getReportOutputDirectory());
659         try {
660             runCheck();
661         } catch (MojoExecutionException ex) {
662             throw new MavenReportException(ex.getMessage(), ex);
663         } catch (MojoFailureException ex) {
664             getLog().warn("Vulnerabilities were identifies that exceed the CVSS threshold for failing the build");
665         }
666     }

680     /**
681      * Returns the correct output directory depending on if a site is being
682      * executed or not.
683      *
684      * @param current the Maven project to get the output directory from
685      * @return the directory to write the report(s)
686      */
687     protected File getCorrectOutputDirectory(MavenProject current) {
688         final Object obj = current.getContextValue(getOutputDirectoryContextKey());
689         if (obj != null && obj instanceof File) {
690             return (File) obj;
691         }
692         File target = new File(current.getBuild().getDirectory());
693         if (target.getParentFile() != null && "target".equals(target.getParentFile().getName())) {
694             target = target.getParentFile();
695         }
696         return target;
697     }

700      * Scans the project's artifacts and adds them to the engine's dependency
701      * list.
702      *
703      * @param project the project to scan the dependencies of
704      * @param engine the engine to use to scan the dependencies
705      * @return a collection of exceptions that may have occurred while resolving
706      * and scanning the dependencies

708     protected ExceptionCollection scanArtifacts(MavenProject project, Engine engine) {
709         try {
710             final DependencyNode dn = dependencyGraphBuilder.buildDependencyGraph(project, null, reactorProjects);
711             final ProjectBuildingRequest buildingRequest = newResolveArtifactProjectBuildingRequest();
712             return collectDependencies(engine, project, dn.getChildren(), buildingRequest);
713         } catch (DependencyGraphBuilderException ex) {

715             getLog().debug(msg, ex);
716             return new ExceptionCollection(msg, ex);
717         }

720     /**
721      * Resolves the projects artifacts using Aether and scans the resulting
722      * dependencies.
723      *
724      * @param engine the core dependency-check engine
725      * @param project the project being scanned
726      * @param nodes the list of dependency nodes, generally obtained via the
727      * DependencyGraphBuilder
728      * @param buildingRequest the Maven project building request
729      * @return a collection of exceptions that may have occurred while resolving
730      * and scanning the dependencies
731      */
732     private ExceptionCollection collectDependencies(Engine engine, MavenProject project,
733             List<DependencyNode> nodes, ProjectBuildingRequest buildingRequest) {
734         ExceptionCollection exCol = null;
735         for (DependencyNode dependencyNode : nodes) {
736             if (artifactScopeExcluded.passes(dependencyNode.getArtifact().getScope())
737                     || artifactTypeExcluded.passes(dependencyNode.getArtifact().getType())) {
738                 continue;
739             }
740             exCol = collectDependencies(engine, project, dependencyNode.getChildren(), buildingRequest);
741             try {
742                 boolean isResolved = false;
743                 File artifactFile = null;
744                 String artifactId = null;
745                 String groupId = null;
746                 String version = null;
747                 List<ArtifactVersion> availableVersions = null;
748                 if (org.apache.maven.artifact.Artifact.SCOPE_SYSTEM.equals(dependencyNode.getArtifact().getScope())) {
749                     for (org.apache.maven.model.Dependency d : project.getDependencies()) {
750                         final Artifact a = dependencyNode.getArtifact();
751                         if (d.getSystemPath() != null && artifactsMatch(d, a)) {
752 
753                             artifactFile = new File(d.getSystemPath());
754                             isResolved = artifactFile.isFile();
755                             groupId = a.getGroupId();
756                             artifactId = a.getArtifactId();
757                             version = a.getVersion();
758                             availableVersions = a.getAvailableVersions();
759                             break;
760                         }
761                     }
762                     if (!isResolved) {
763                         getLog().error("Unable to resolve system scoped dependency: " + dependencyNode.toNodeString());
764                         if (exCol == null) {
765                             exCol = new ExceptionCollection();
766                         }
767                         exCol.addException(new DependencyNotFoundException("Unable to resolve system scoped dependency: "
768                                 + dependencyNode.toNodeString()));
769                     }
770                 } else {
771                     final ArtifactCoordinate coordinate = TransferUtils.toArtifactCoordinate(dependencyNode.getArtifact());
772                     final Artifact result = artifactResolver.resolveArtifact(buildingRequest, coordinate).getArtifact();
773                     isResolved = result.isResolved();
774                     artifactFile = result.getFile();
775                     groupId = result.getGroupId();
776                     artifactId = result.getArtifactId();
777                     version = result.getVersion();
778                     availableVersions = result.getAvailableVersions();
779                 }
780                 if (isResolved && artifactFile != null) {
781                     final List<Dependency> deps = engine.scan(artifactFile.getAbsoluteFile(),
782                             project.getName() + ":" + dependencyNode.getArtifact().getScope());
783                     if (deps != null) {
784                         if (deps.size() == 1) {
785                             final Dependency d = deps.get(0);
786                             if (d != null) {
787                                 final MavenArtifact ma = new MavenArtifact(groupId, artifactId, version);
788                                 d.addAsEvidence("pom", ma, Confidence.HIGHEST);
789                                 if (availableVersions != null) {
790                                     for (ArtifactVersion av : availableVersions) {
791                                         d.addAvailableVersion(av.toString());
792                                     }
793                                 }
794                                 getLog().debug(String.format("Adding project reference %s on dependency %s",
795                                         project.getName(), d.getDisplayFileName()));
796                             }
797                         } else if (getLog().isDebugEnabled()) {
798                             final String msg = String.format("More than 1 dependency was identified in first pass scan of '%s' in project %s",
799                                     dependencyNode.getArtifact().getId(), project.getName());
800                             getLog().debug(msg);
801                         }
802                     } else if ("import".equals(dependencyNode.getArtifact().getScope())) {
803                         final String msg = String.format("Skipping '%s:%s' in project %s as it uses an `import` scope",
804                                 dependencyNode.getArtifact().getId(), dependencyNode.getArtifact().getScope(), project.getName());
805                         getLog().debug(msg);
806                     } else {
807                         final String msg = String.format("No analyzer could be found for '%s:%s' in project %s",
808                                 dependencyNode.getArtifact().getId(), dependencyNode.getArtifact().getScope(), project.getName());
809                         getLog().warn(msg);
810                     }
811                 } else {
812                     final String msg = String.format("Unable to resolve '%s' in project %s",
813                             dependencyNode.getArtifact().getId(), project.getName());
814                     getLog().debug(msg);
815                     if (exCol == null) {
816                         exCol = new ExceptionCollection();
817                     }
818                 }
819             } catch (ArtifactResolverException ex) {
820                 if (exCol == null) {
821                     exCol = new ExceptionCollection();
822                 }
823                 exCol.addException(ex);
824             }
825         }
826 
827         // Define the default FileSets
828         if (scanSet == null || scanSet.length == 0) {
829             final FileSet resourcesSet = new FileSet();
830             final FileSet filtersSet = new FileSet();
831             final FileSet webappSet = new FileSet();
832             try {
833                 resourcesSet.setDirectory(new File(project.getBasedir(), "src/main/resources").getCanonicalPath());
834                 filtersSet.setDirectory(new File(project.getBasedir(), "src/main/filters").getCanonicalPath());
835                 webappSet.setDirectory(new File(project.getBasedir(), "src/main/webapp").getCanonicalPath());
836             } catch (IOException ex) {
837                 if (exCol == null) {
838                     exCol = new ExceptionCollection();
839                 }
840                 exCol.addException(ex);
841             }
842             scanSet = new FileSet[]{resourcesSet, filtersSet, webappSet};
843         }
844         // Iterate through FileSets and scan included files
845         final FileSetManager fileSetManager = new FileSetManager();
846         for (FileSet fileSet : scanSet) {
847             final String[] includedFiles = fileSetManager.getIncludedFiles(fileSet);
848             for (String include : includedFiles) {
849                 final File includeFile = new File(fileSet.getDirectory(), include).getAbsoluteFile();
850                 if (includeFile.exists()) {
851                     engine.scan(includeFile, project.getName());
852                 }
853                 //TODO - should we add an exception/error reporting for files that do not exist?
854             }
855         }
856 
857         return exCol;
858     }

865      * @param a the Maven artifact

871                 && (isEqualOrNull(a.getVersion(), d.getVersion()));

877      *

898     /**
899      * Executes the dependency-check scan and generates the necessary report.
900      *
901      * @throws MojoExecutionException thrown if there is an exception running
902      * the scan
903      * @throws MojoFailureException thrown if dependency-check is configured to
904      * fail the build
905      */
906     protected void runCheck() throws MojoExecutionException, MojoFailureException {
907         try (Engine engine = initializeEngine()) {
908             ExceptionCollection exCol = scanDependencies(engine);
909             try {
910                 engine.analyzeDependencies();
911             } catch (ExceptionCollection ex) {
912                 exCol = handleAnalysisExceptions(exCol, ex);
913             }
914             if (exCol == null || !exCol.isFatal()) {
915 
916                 File outputDir = getCorrectOutputDirectory(this.getProject());
917                 if (outputDir == null) {
918                     //in some regards we shouldn't be writing this, but we are anyway.
919                     //we shouldn't write this because nothing is configured to generate this report.
920                     outputDir = new File(this.getProject().getBuild().getDirectory());
921                 }
922                 try {
923                     final MavenProject p = this.getProject();
924                     engine.writeReports(p.getName(), p.getGroupId(), p.getArtifactId(), p.getVersion(), outputDir, getFormat());
925                 } catch (ReportException ex) {
926                     if (exCol == null) {
927                         exCol = new ExceptionCollection("Error writing aggregate report", ex);
928                     } else {
929                         exCol.addException(ex);
930                     }
931                     if (this.isFailOnError()) {
932                         throw new MojoExecutionException("One or more exceptions occurred during dependency-check analysis", exCol);
933                     } else {
934                         getLog().debug("Error writting the report", ex);
935                     }
936                 }
937                 showSummary(this.getProject(), engine.getDependencies());
938                 checkForFailure(engine.getDependencies());
939                 if (exCol != null && this.isFailOnError()) {
940                     throw new MojoExecutionException("One or more exceptions occurred during dependency-check analysis", exCol);
941                 }
942             }
943         } catch (DatabaseException ex) {
944             if (getLog().isDebugEnabled()) {
945                 getLog().debug("Database connection error", ex);
946             }
947             final String msg = "An exception occurred connecting to the local database. Please see the log file for more details.";
948             if (this.isFailOnError()) {
949                 throw new MojoExecutionException(msg, ex);
950             }
951             getLog().error(msg, ex);
952         } finally {
953             getSettings().cleanup();
954         }
955     }

957     /**
958      * Combines the two exception collections and if either are fatal, throw an
959      * MojoExecutionException
960      *
961      * @param currentEx the primary exception collection
962      * @param newEx the new exception collection to add
963      * @return the combined exception collection
964      * @throws MojoExecutionException
965      */
966     private ExceptionCollection handleAnalysisExceptions(ExceptionCollection currentEx, ExceptionCollection newEx) throws MojoExecutionException {
967         ExceptionCollection returnEx = currentEx;
968         if (returnEx == null) {
969             returnEx = newEx;
970         } else {
971             returnEx.getExceptions().addAll(newEx.getExceptions());
972             if (newEx.isFatal()) {
973                 returnEx.setFatal(true);
974             }
975         }
976         if (returnEx.isFatal()) {
977             final String msg = String.format("Fatal exception(s) analyzing %s", getProject().getName());
978             if (this.isFailOnError()) {
979                 throw new MojoExecutionException(msg, returnEx);
980             }
981             getLog().error(msg);
982             if (getLog().isDebugEnabled()) {
983                 getLog().debug(returnEx);
984             }
985         } else {
986             final String msg = String.format("Exception(s) analyzing %s", getProject().getName());
987             if (getLog().isDebugEnabled()) {
988                 getLog().debug(msg, returnEx);
989             }
990         }
991         return returnEx;
992     }

1251     /**
1252      * Returns the maven proxy.
1253      *
1254      * @return the maven proxy
1255      */
1256     private Proxy getMavenProxy() {
1257         if (mavenSettings != null) {
1258             final List<Proxy> proxies = mavenSettings.getProxies();
1259             if (proxies != null && !proxies.isEmpty()) {
1260                 if (mavenSettingsProxyId != null) {
1261                     for (Proxy proxy : proxies) {
1262                         if (mavenSettingsProxyId.equalsIgnoreCase(proxy.getId())) {
1263                             return proxy;
1264                         }
1265                     }
1266                 } else if (proxies.size() == 1) {
1267                     return proxies.get(0);
1268                 } else {
1269                     getLog().warn("Multiple proxy definitions exist in the Maven settings. In the dependency-check "
1270                             + "configuration set the mavenSettingsProxyId so that the correct proxy will be used.");
1271                     throw new IllegalStateException("Ambiguous proxy definition");

1329     /**
1330      * Checks to see if a vulnerability has been identified with a CVSS score
1331      * that is above the threshold set in the configuration.
1332      *
1333      * @param dependencies the list of dependency objects
1334      * @throws MojoFailureException thrown if a CVSS score is found that is
1335      * higher then the threshold set
1336      */
1337     protected void checkForFailure(Dependency[] dependencies) throws MojoFailureException {
1338         final StringBuilder ids = new StringBuilder();
1339         for (Dependency d : dependencies) {
1340             boolean addName = true;
1341             for (Vulnerability v : d.getVulnerabilities()) {
1342                 if (failBuildOnAnyVulnerability || v.getCvssScore() >= failBuildOnCVSS) {
1343                     if (addName) {
1344                         addName = false;
1345                         ids.append(NEW_LINE).append(d.getFileName()).append(": ");
1346                         ids.append(v.getName());
1347                     } else {
1348                         ids.append(", ").append(v.getName());
1349                     }
1350                 }
1351             }
1352         }
1353         if (ids.length() > 0) {
1354             final String msg;
1355             if (showSummary) {
1356                 if (failBuildOnAnyVulnerability) {
1357                     msg = String.format("%n%nOne or more dependencies were identified with vulnerabilities: %n%s%n%n"
1358                             + "See the dependency-check report for more details.%n%n", ids.toString());
1359                 } else {
1360                     msg = String.format("%n%nOne or more dependencies were identified with vulnerabilities that have a CVSS score greater than or equal to '%.1f': "
1361                             + "%n%s%n%nSee the dependency-check report for more details.%n%n", failBuildOnCVSS, ids.toString());
1362                 }
1363             } else {
1364                 msg = String.format("%n%nOne or more dependencies were identified with vulnerabilities.%n%n"
1365                         + "See the dependency-check report for more details.%n%n");
1366             }
1367 
1368             throw new MojoFailureException(msg);
1369         }
1370     }

1372     /**
1373      * Generates a warning message listing a summary of dependencies and their
1374      * associated CPE and CVE entries.
1375      *
1376      * @param mp the Maven project for which the summary is shown
1377      * @param dependencies a list of dependency objects
1378      */
1379     protected void showSummary(MavenProject mp, Dependency[] dependencies) {
1380         if (showSummary) {
1381             final StringBuilder summary = new StringBuilder();
1382             for (Dependency d : dependencies) {
1383                 boolean firstEntry = true;
1384                 final StringBuilder ids = new StringBuilder();
1385                 for (Vulnerability v : d.getVulnerabilities()) {
1386                     if (firstEntry) {
1387                         firstEntry = false;
1388                     } else {
1389                         ids.append(", ");
1390                     }
1391                     ids.append(v.getName());
1392                 }
1393                 if (ids.length() > 0) {
1394                     summary.append(d.getFileName()).append(" (");
1395                     firstEntry = true;
1396                     for (Identifier id : d.getIdentifiers()) {
1397                         if (firstEntry) {
1398                             firstEntry = false;
1399                         } else {
1400                             summary.append(", ");
1401                         }
1402                         summary.append(id.getValue());
1403                     }
1404                     summary.append(") : ").append(ids).append(NEW_LINE);
1405                 }
1406             }
1407             if (summary.length() > 0) {
1408                 final String msg = String.format("%n%n" + "One or more dependencies were identified with known vulnerabilities in %s:%n%n%s"
1409                         + "%n%nSee the dependency-check report for more details.%n%n", mp.getName(), summary.toString());
1410                 getLog().warn(msg);
1411             }
1412         }
1413     }
