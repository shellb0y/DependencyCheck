18  package org.owasp.dependencycheck.utils;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.io.RandomAccessFile;
23  import java.nio.channels.FileLock;
24  import java.security.SecureRandom;
25  import java.sql.Timestamp;
26  import java.util.Date;
27  import javax.annotation.concurrent.NotThreadSafe;
28  import org.owasp.dependencycheck.exception.H2DBLockException;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  /**
33   *
34   * @author Jeremy Long
35   */
36  @NotThreadSafe
37  public class H2DBLock {
38  
39      /**
40       * The logger.
41       */
42      private static final Logger LOGGER = LoggerFactory.getLogger(H2DBLock.class);
43      /**
44       * How long to sleep waiting for the lock.
45       */
46      public static final int SLEEP_DURATION = 10000;
47      /**
48       * Max attempts to obtain a lock.
49       */
50      public static final int MAX_SLEEP_COUNT = 120;
51      /**
52       * The file lock.
53       */
54      private FileLock lock = null;
55      /**
56       * Reference to the file that we are locking.
57       */
58      private RandomAccessFile file = null;
59      /**
60       * The lock file.
61       */
62      private File lockFile = null;
63      /**
64       * The configured settings.
65       */
66      private final Settings settings;
67      /**
68       * A random string used to validate the lock.
69       */
70      private final String magic;
71  
72      /**
73       * Constructs a new H2DB Lock object with the configured settings.
74       *
75       * @param settings the configured settings
76       */
77      public H2DBLock(Settings settings) {
78          this.settings = settings;
79          final byte[] random = new byte[16];
80          final SecureRandom gen = new SecureRandom();
81          gen.nextBytes(random);
82          magic = Checksum.getHex(random);
83      }
84  
85      /**
86       * Determine if the lock is currently held.
87       *
88       * @return true if the lock is currently held
89       */
90      public boolean isLocked() {
91          return lock != null && lock.isValid();
92      }
93  
94      /**
95       * Obtains a lock on the H2 database.
96       *
97       * @throws H2DBLockException thrown if a lock could not be obtained
98       */
99      public void lock() throws H2DBLockException {
100         try {
101             final File dir = settings.getDataDirectory();
102             lockFile = new File(dir, "dc.update.lock");
103             if (!lockFile.getParentFile().isDirectory() && !lockFile.mkdir()) {
104                 throw new H2DBLockException("Unable to create path to data directory.");
105             }
106             if (lockFile.isFile() && getFileAge(lockFile) > 30) {
107                 LOGGER.debug("An old db update lock file was found: {}", lockFile.getAbsolutePath());
108                 if (!lockFile.delete()) {
109                     LOGGER.warn("An old db update lock file was found but the system was unable to delete "
110                             + "the file. Consider manually deleting {}", lockFile.getAbsolutePath());
111                 }
112             }
113             int ctr = 0;
114             do {
115                 try {
116                     if (!lockFile.exists() && lockFile.createNewFile()) {
117                         file = new RandomAccessFile(lockFile, "rw");
118                         lock = file.getChannel().lock();
119                         file.writeBytes(magic);
120                         file.getChannel().force(true);
121                         Thread.sleep(20);
122                         file.seek(0);
123                         final String current = file.readLine();
124                         if (current != null && !current.equals(magic)) {
125                             lock.close();
126                             lock = null;
127                             LOGGER.debug("Another process obtained a lock first ({})", Thread.currentThread().getName());
128                         } else {
129                             final Timestamp timestamp = new Timestamp(System.currentTimeMillis());
130                             LOGGER.debug("Lock file created ({}) {} @ {}", Thread.currentThread().getName(), magic, timestamp.toString());
131                         }
132                     }
133                 } catch (IOException | InterruptedException ex) {
134                     LOGGER.trace("Expected error as another thread has likely locked the file", ex);
135                 } finally {
136                     if (lock == null && file != null) {
137                         try {
138                             file.close();
139                             file = null;
140                         } catch (IOException ex) {
141                             LOGGER.trace("Unable to close the lock file", ex);
142                         }
143                     }
144                 }
145                 if (lock == null || !lock.isValid()) {
146                     try {
147                         final Timestamp timestamp = new Timestamp(System.currentTimeMillis());
148                         LOGGER.debug("Sleeping thread {} ({}) for 10 seconds because an exclusive lock on the database could not be obtained ({})",
149                                 Thread.currentThread().getName(), magic, timestamp.toString());
150                         Thread.sleep(SLEEP_DURATION);
151                     } catch (InterruptedException ex) {
152                         LOGGER.debug("sleep was interrupted.", ex);
153                         Thread.currentThread().interrupt();
154                     }
155                 }
156             } while (++ctr < MAX_SLEEP_COUNT && (lock == null || !lock.isValid()));
157             if (lock == null || !lock.isValid()) {
158                 throw new H2DBLockException("Unable to obtain the update lock, skipping the database update. Skippinig the database update.");
159             }
160         } catch (IOException ex) {
161             throw new H2DBLockException(ex.getMessage(), ex);
162         }
163     }
164 
165     /**
166      * Releases the lock on the H2 database.
167      */
168     public void release() {
169         if (lock != null) {
170             try {
171                 lock.release();
172                 lock = null;
173             } catch (IOException ex) {
174                 LOGGER.debug("Failed to release lock", ex);
175             }
176         }
177         if (file != null) {
178             try {
179                 file.close();
180                 file = null;
181             } catch (IOException ex) {
182                 LOGGER.debug("Unable to delete lock file", ex);
183             }
184         }
185         if (lockFile != null && lockFile.isFile()) {
186             try (RandomAccessFile f = new RandomAccessFile(lockFile, "rw")) {
187                 final String m = f.readLine();
188                 //yes, we are explicitly calling close on an auto-closable object - this is so we can delete the file.
189                 f.close();
190                 if (m != null && m.equals(magic) && !lockFile.delete()) {
191                     LOGGER.error("Lock file '{}' was unable to be deleted. Please manually delete this file.", lockFile.toString());
192                     lockFile.deleteOnExit();
193                 }
194             } catch (IOException ex) {
195                 LOGGER.debug("Error deleting lock file", ex);
196             }
197         }
198         lockFile = null;
199         final Timestamp timestamp = new Timestamp(System.currentTimeMillis());
200         LOGGER.debug("Lock released ({}) {} @ {}", Thread.currentThread().getName(), magic, timestamp.toString());
201     }
202 
203     /**
204      * Returns the age of the file in minutes.
205      *
206      * @param file the file to calculate the age
207      * @return the age of the file
208      */
209     private double getFileAge(File file) {
210         final Date d = new Date();
211         final long modified = file.lastModified();
212         final double time = (d.getTime() - modified) / 1000.0 / 60.0;
213         LOGGER.debug("Lock file age is {} minutes", time);
214         return time;
215     }
216 }
